    export const getGridStatsForStrategiesAction = async (
      gridIds: string[]
    ): Promise<ApiResponse<Record<string, { totalBuys: number; totalSells: number; realizedPnl: number }>>> => {
      try {
        const userId = await getCurrentUserId()
        const service = getService()
        const results: Record<string, { totalBuys: number; totalSells: number; realizedPnl: number }> = {}
        const settled = await Promise.allSettled(
