    2. Create `src/components/portfolio/instrument-chart.tsx`:
       - `"use client"` directive
       - Import proper types from lightweight-charts for type safety (NO `as any` casts):
         ```typescript
         import { createChart, type IChartApi, type CandlestickData, type SeriesMarker, type Time, ColorType } from "lightweight-charts"
         ```
       - `type InstrumentChartProps = { candles: CandlestickData<Time>[]; markers?: SeriesMarker<Time>[]; height?: number }`
       - Implementation using useRef + useEffect:
         ```typescript
         const containerRef = useRef<HTMLDivElement>(null)
         const chartRef = useRef<IChartApi | null>(null)

         useEffect(() => {
           if (!containerRef.current || candles.length === 0) return

           const chart = createChart(containerRef.current, {
             layout: {
               background: { type: ColorType.Solid, color: "transparent" },
               textColor: "hsl(var(--foreground))",
             },
             grid: {
               vertLines: { color: "hsl(var(--border))" },
               horzLines: { color: "hsl(var(--border))" },
             },
             width: containerRef.current.clientWidth,
             height: height ?? 400,
             autoSize: true,
             rightPriceScale: { borderColor: "hsl(var(--border))" },
             timeScale: { borderColor: "hsl(var(--border))" },
             crosshair: {
               horzLine: { color: "hsl(var(--muted-foreground))", labelBackgroundColor: "hsl(var(--card))" },
               vertLine: { color: "hsl(var(--muted-foreground))", labelBackgroundColor: "hsl(var(--card))" },
             },
           })

           const series = chart.addCandlestickSeries({
             upColor: "#10b981",
             downColor: "#ef4444",
             borderVisible: false,
             wickUpColor: "#10b981",
             wickDownColor: "#ef4444",
           })

           series.setData(candles)
           if (markers?.length) {
             const sorted = [...markers].sort((a, b) =>
               String(a.time).localeCompare(String(b.time))
             )
             series.setMarkers(sorted)
           }
           chart.timeScale().fitContent()
           chartRef.current = chart

           return () => { chart.remove(); chartRef.current = null }
         }, [candles, markers, height])

         return <div ref={containerRef} className="w-full" />
         ```
       - CRITICAL: Use `CandlestickData<Time>[]` for setData and `SeriesMarker<Time>[]` for setMarkers — do NOT use `as any`. The ChartCandle type from chart-actions.ts must match `CandlestickData<Time>` shape (time, open, high, low, close).
       - IMPORTANT: This component must be loaded via `next/dynamic({ ssr: false })` by consumers.
       - Named export: `export const InstrumentChart`

    NOTE on reusability (CHRT-07): InstrumentChart accepts `candles` and `markers` props — it does NOT fetch data itself. Consumers (terminal page, portfolio expand) provide data via server actions. This makes it reusable across contexts.
