// Backtest singleton — use Backtest.run() async generator
declare const Backtest: BacktestUtils;

// Backtest.run() returns async generator of signal results
run: (symbol: string, context: {
  strategyName: StrategyName;
  exchangeName: ExchangeName;
  frameName: FrameName;
}) => AsyncGenerator<IStrategyTickResultScheduled | IStrategyTickResultOpened | IStrategyTickResultClosed | IStrategyTickResultCancelled, void, any>;

// Backtest.getData() returns stats after run completes
getData: (symbol: string, context: {
  strategyName: StrategyName;
  exchangeName: ExchangeName;
  frameName: FrameName;
}) => Promise<BacktestStatisticsModel>;

interface BacktestStatisticsModel {
  totalSignals: number;
  winCount: number;
  lossCount: number;
  winRate: number | null;
  avgPnl: number | null;
  totalPnl: number | null;
  sharpeRatio: number | null;
  signalList: IStrategyTickResultClosed[];
}

// Strategy schema registration
declare function addStrategySchema(strategySchema: IStrategySchema): void;

interface IStrategySchema {
  strategyName: StrategyName;
  interval: SignalInterval; // "1m" | "3m" | "5m" | "15m" | "30m" | "1h"
  getSignal: (symbol: string, when: Date) => Promise<ISignalDto | null>;
  callbacks?: Partial<IStrategyCallbacks>;
}

interface ISignalDto {
  position: "long" | "short";
  priceOpen?: number;
  priceTakeProfit: number;
  priceStopLoss: number;
  minuteEstimatedTime?: number;
}
